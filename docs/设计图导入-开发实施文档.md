## 三、app_ui.py：新增图片解析函数

在 `_generate_entry_summary` 函数附近（约 115 行），新增以下函数：

```python
def _parse_design_image_with_gemini(
    image_bytes: bytes,
    mime_type: str,
    gemini_key: str = "",
) -> tuple[str, str]:
    """
    调用 Gemini Vision 解析设计图，返回结构化文本描述。
    返回 (description_text, error_message)。
    error_message 为空字符串表示成功。
    """
    key = (gemini_key or "").strip() or os.environ.get("GEMINI_API_KEY", "")
    if not key:
        return "", "未配置 Gemini API Key"
    if not image_bytes:
        return "", "图片内容为空"

    DESIGN_VISION_PROMPT = (
        "你是一位资深 UI/UX 测试工程师。请分析这张界面设计图，输出结构化描述，"
        "供自动化测试用例生成系统参考。\n\n"
        "输出格式：\n"
        "## 页面 / 模块名称\n（根据图中标题或推断得出）\n\n"
        "## 页面层级与布局\n（顶栏、底栏、侧栏、主内容区大体布局）\n\n"
        "## 关键 UI 组件\n（按钮、输入框、列表项、弹窗等，说明标签文案和状态）\n\n"
        "## 可见的交互入口\n（Tab 切换、下拉菜单、长按操作等）\n\n"
        "## 状态与条件展示\n（空态、加载态、错误态、权限不足态等，如图中有体现）\n\n"
        "## 关键文案\n（图中重要的 UI 文案、错误提示、占位符文字等）\n\n"
        "只输出上述结构，不加多余说明。"
    )

    try:
        import base64
        from langchain_core.messages import HumanMessage

        raw_model = os.environ.get("GEMINI_MODEL", "gemini-2.5-flash-lite")
        model = _resolve_gemini_model(raw_model)

        from langchain_google_genai import ChatGoogleGenerativeAI
        llm = ChatGoogleGenerativeAI(model=model, google_api_key=key, temperature=0.1)

        b64 = base64.b64encode(image_bytes).decode("utf-8")
        message = HumanMessage(content=[
            {"type": "image_url", "image_url": {"url": f"data:{mime_type};base64,{b64}"}},
            {"type": "text", "text": DESIGN_VISION_PROMPT},
        ])
        resp = llm.invoke([message])
        text = (resp.content or "").strip()
        if not text:
            return "", "Gemini 返回内容为空，请确认图片清晰度"
        return text, ""
    except Exception as ex:
        err = str(ex)
        return "", f"解析失败：{err}"
```

**注意事项**：
- `HumanMessage` 的图片格式为 `data:{mime_type};base64,{...}`，langchain_google_genai ≥ 0.1.x 均支持。
- 函数内不写任何文件 IO，图片字节不落盘。
- 所有异常由 try-except 捕获，对上层只返回错误字符串，不抛异常。

## 四、app_ui.py：UI 区块

在 `render_memory_tab` 函数中，「导入全回归测试用例」区块结束后、「项目记忆摘要（高级）」expander 前，插入以下代码：

```python
# === 导入设计图 ===
st.markdown("**" + (_get_text(T, "memory_tab.design_mockup_section") or "导入设计图") + "**")
st.caption(_get_text(T, "memory_tab.design_mockup_caption") or "上传 Figma/截图等设计稿，Agent 将理解界面布局与交互细节。")

try:
    from app_ui_components import render_file_uploader
    design_upload_result = render_file_uploader(
        accepted_types=["png", "jpg", "jpeg", "pdf"],
        key="design_mockup_upload",
        label=_get_text(T, "memory_tab.design_mockup_upload_label") or "上传设计图（PNG / JPG / PDF，最多 5 个，单文件 ≤10MB）",
        multiple=True,  # 若 render_file_uploader 支持 multiple
    )
    design_files = design_upload_result.get("files") if design_upload_result else None
except ImportError:
    design_files = st.file_uploader(
        _get_text(T, "memory_tab.design_mockup_upload_label") or "上传设计图（PNG / JPG / PDF，最多 5 个，单文件 ≤10MB）",
        type=["png", "jpg", "jpeg", "pdf"],
        key="design_mockup_upload",
        accept_multiple_files=True,
        label_visibility="collapsed",
    )

if st.button(
    _get_text(T, "memory_tab.design_mockup_import_btn") or "解析并导入",
    key="mem_import_design",
):
    gemini_key = defaults.get("gemini_key", "")
    if not gemini_key:
        st.error(_get_text(T, "memory_tab.design_mockup_key_missing") or "请先在设置中配置 Gemini API Key")
    elif not design_files:
        st.error(_get_text(T, "memory_tab.import_required") or "请上传设计图文件")
    else:
        files_to_process = design_files[:5]  # 最多 5 个
        if len(design_files) > 5:
            st.warning("已超过 5 个文件限制，只处理前 5 个。")
        
        import_count = 0
        skip_count = 0
        fail_msgs = []

        for uploaded_file in files_to_process:
            file_name = getattr(uploaded_file, "name", "设计图")
            raw_bytes = uploaded_file.read()

            # 文件大小检查（10MB）
            if len(raw_bytes) > 10 * 1024 * 1024:
                fail_msgs.append(f"{file_name}：文件超过 10MB，已跳过")
                continue

            # 计算哈希去重
            import hashlib
            file_hash = hashlib.sha256(raw_bytes).hexdigest()
            # 用哈希作为 source_id 实现去重（add_entry_with_dedup 内部也有 content_hash 去重，
            # 但这里的去重基于原始图片字节，与存储的文本内容无关，故以 source_id 区分）

            ext = file_name.lower().rsplit(".", 1)[-1] if "." in file_name else "png"
            mime_map = {"jpg": "image/jpeg", "jpeg": "image/jpeg", "png": "image/png", "pdf": "application/pdf"}
            mime_type = mime_map.get(ext, "image/png")

            # PDF 分页处理（最多 5 页）
            pages_bytes: list[tuple[bytes, str]] = []
            if ext == "pdf":
                try:
                    import fitz  # PyMuPDF，尝试导入
                    from io import BytesIO
                    doc = fitz.open(stream=raw_bytes, filetype="pdf")
                    total_pages = len(doc)
                    if total_pages > 5:
                        st.info(
                            (_get_text(T, "memory_tab.design_mockup_pdf_truncated") or "PDF 超过 5 页，已只处理前 5 页")
                            + f"（共 {total_pages} 页）"
                        )
                    for i in range(min(total_pages, 5)):
                        page = doc[i]
                        pix = page.get_pixmap(dpi=150)
                        img_bytes = pix.tobytes("png")
                        pages_bytes.append((img_bytes, "image/png"))
                    doc.close()
                except ImportError:
                    # PyMuPDF 不可用时，直接把 PDF 原始字节发给 Gemini（Gemini 本身支持 PDF）
                    pages_bytes = [(raw_bytes, "application/pdf")]
                except Exception as ex:
                    fail_msgs.append(f"{file_name}：PDF 解析失败 - {ex}")
                    continue
            else:
                pages_bytes = [(raw_bytes, mime_type)]

            # 逐页/逐图解析（串行，每图间隔 1s 防限速）
            page_descriptions = []
            import time
            for idx, (pb, pm) in enumerate(pages_bytes):
                with st.spinner(
                    (_get_text(T, "memory_tab.design_mockup_parsing") or "正在用 Gemini Vision 解析设计图…")
                    + (f"（第 {idx + 1} 页）" if len(pages_bytes) > 1 else "")
                ):
                    desc, err = _parse_design_image_with_gemini(pb, pm, gemini_key)
                if err:
                    fail_msgs.append(f"{file_name} 第{idx+1}页：{err}")
                else:
                    page_descriptions.append(f"### 第 {idx+1} 页\n\n{desc}" if len(pages_bytes) > 1 else desc)
                if idx < len(pages_bytes) - 1:
                    time.sleep(1)

            if not page_descriptions:
                continue

            full_description = f"# 设计图：{file_name}\n\n" + "\n\n---\n\n".join(page_descriptions)

            # 入库（以文本内容哈希去重）
            rowid, status = add_entry_with_dedup(
                DESIGN_MOCKUP_SOURCE_TYPE,
                full_description,
                source_id=file_hash[:16],  # 用图片哈希前16位作为唯一标识
                title=f"设计图：{file_name}",
                summary=full_description[:500],
            )

            if status == "skipped":
                skip_count += 1
            else:
                import_count += 1
                try:
                    from context_cache_service import mark_context_cache_dirty
                    mark_context_cache_dirty(f"design_{status}")
                except ImportError:
                    pass
                # 生成 Librarian 摘要
                with st.spinner(_get_text(T, "memory_tab.agent_summary_pending") or "生成摘要中…"):
                    _generate_entry_summary(rowid, full_description, gemini_key)

        # 汇总结果展示
        if import_count:
            st.success(
                (_get_text(T, "memory_tab.design_mockup_success") or "设计图已解析导入，下次生成用例时 Agent 将参考。")
                + f"（{import_count} 个）"
            )
        if skip_count:
            st.info(
                (_get_text(T, "memory_tab.design_mockup_skipped") or "设计图未变更，已跳过")
                + f"（{skip_count} 个）"
            )
        for msg in fail_msgs:
            st.error(msg)
        if import_count or skip_count:
            st.rerun()
```

## 七、关于 PyMuPDF（PDF 分页）

PDF 分页依赖 `PyMuPDF`（`import fitz`），**Streamlit Cloud 默认不安装**。

处理策略：
1. 代码中用 `try: import fitz ... except ImportError: ...` 包裹。
2. **ImportError 时的降级**：把 PDF 原始字节直接发给 Gemini（Gemini API 原生支持 PDF，无需拆页）。
3. `requirements.txt` 中**不强制加入** PyMuPDF，作为可选增强。

```python
# requirements.txt 注释说明（不强制安装）
# PyMuPDF  # 可选：PDF 分页处理，不安装时 PDF 直接整体发给 Gemini
```

## 九、验证步骤（上线前必测）

1. **本地测试**：上传 1 张 PNG，确认 Gemini Vision 返回结构化描述，导入历史出现条目，source_type = design_mockup。
2. **去重测试**：同一张图再次上传，提示"已跳过"，历史条数不增加。
3. **无 API Key 测试**：清空 Gemini Key，点击导入，st.error 出现，不崩溃。
4. **PDF 测试**：上传多页 PDF（fitz 可用），确认前 5 页解析并合并入库。
5. **PDF 无 fitz 测试**：注释掉 fitz import，上传 PDF，确认降级为整体发送 Gemini，正常入库。
6. **sqlite3 不可用测试**：mock sqlite3 缺失，确认切换 JSON 后端后，design_mockup 导入正常。
7. **Agent 上下文测试**：导入设计图后，触发一次生成用例，检查 project_context 中是否包含"设计图"相关文字。
